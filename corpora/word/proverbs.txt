אֵ֣י שֵׁכָֽר׃ eov eos פֶּן־ יִ֭שְׁתֶּה eos וְ יִשְׁכַּ֣ח מְחֻקָּ֑ק eos וִֽ֝ ישַׁנֶּה דִּ֣ין כָּל־ בְּנֵי־ עֹֽנִי׃ eov eos תְּנוּ־ שֵׁכָ֣ר לְ אֹובֵ֑ד וְ֝ יַיִן לְ מָ֣רֵי נָֽפֶשׁ׃ eov eos יִ֭שְׁתֶּה eos וְ יִשְׁכַּ֣ח רִישֹׁ֑ו eos וַ֝ עֲמָלֹ֗ו לֹ֣א יִזְכָּר־ עֹֽוד׃ eov eos פְּתַח־ פִּ֥יךָ לְ אִלֵּ֑ם אֶל־ דִּ֝֗ין כָּל־ בְּנֵ֥י חֲלֹֽוף׃ eov eos פְּתַח־ פִּ֥יךָ eos שְׁפָט־ צֶ֑דֶק eos וְ֝ דִ֗ין עָנִ֥י וְ אֶבְיֹֽון׃ פ eov eos אֵֽשֶׁת־ חַ֭יִל מִ֣י יִמְצָ֑א eos וְ רָחֹ֖ק מִ פְּנִינִ֣ים מִכְרָֽהּ׃ eov eos בָּ֣טַח בָּ֭הּ לֵ֣ב בַּעְלָ֑הּ eos וְ֝ שָׁלָ֗ל לֹ֣א יֶחְסָֽר׃ eov eos גְּמָלַ֣תְהוּ טֹ֣וב וְ לֹא־ רָ֑ע כֹּ֝֗ל יְמֵ֣י חַיֶּֽיה׃ eov eos דָּ֭רְשָׁה צֶ֣מֶר וּ פִשְׁתִּ֑ים eos וַ֝ תַּ֗עַשׂ בְּ חֵ֣פֶץ כַּפֶּֽיהָ׃ eov eos הָ֭יְתָה כָּ אֳנִיֹּ֣ות סֹוחֵ֑ר eos מִ֝ מֶּרְחָ֗ק תָּבִ֥יא לַחְמָֽהּ׃ eov eos וַ תָּ֤קָם׀ בְּ עֹ֬וד לַ֗יְלָה eos וַ תִּתֵּ֣ן טֶ֣רֶף לְ בֵיתָ֑הּ וְ֝ חֹ֗ק לְ נַעֲרֹתֶֽיהָ׃ eov eos זָמְמָ֣ה שָׂ֭דֶה eos וַ תִּקָּחֵ֑הוּ eos מִ פְּרִ֥י כַ֝פֶּ֗יהָ נָ֣טְעָה כָּֽרֶם׃ eov eos חָֽגְרָ֣ה בְ עֹ֣וז מָתְנֶ֑יהָ eos וַ֝ תְּאַמֵּ֗ץ זְרֹעֹותֶֽיהָ׃ eov eos טָ֭עֲמָה כִּי־ טֹ֣וב סַחְרָ֑הּ eos לֹֽא־ יִכְבֶּ֖ה בַ לַּ֣יְלָה נֵרָֽהּ׃ eov eos יָ֭דֶיהָ שִׁלְּחָ֣ה בַ כִּישֹׁ֑ור eos וְ֝ כַפֶּ֗יהָ תָּ֣מְכוּ פָֽלֶךְ׃ eov eos כַּ֭פָּהּ פָּרְשָׂ֣ה לֶ עָנִ֑י eos וְ֝ יָדֶ֗יהָ שִׁלְּחָ֥ה לָֽ אֶבְיֹֽון׃ eov eos לֹא־ תִירָ֣א לְ בֵיתָ֣הּ מִ שָּׁ֑לֶג eos כִּ֥י כָל־ בֵּ֝יתָ֗הּ לָבֻ֥שׁ שָׁנִֽים׃ eov eos מַרְבַדִּ֥ים עָֽשְׂתָה־ לָּ֑הּ eos שֵׁ֖שׁ וְ אַרְגָּמָ֣ן לְבוּשָֽׁהּ׃ eov eos נֹודָ֣ע בַּ שְּׁעָרִ֣ים בַּעְלָ֑הּ בְּ֝ שִׁבְתֹּ֗ו עִם־ זִקְנֵי־ אָֽרֶץ׃ eov eos סָדִ֣ין עָ֭שְׂתָה eos וַ תִּמְכֹּ֑ר eos וַ֝ חֲגֹ֗ור נָתְנָ֥ה לַֽ כְּנַעֲנִֽי׃ eov eos עֹז־ וְ הָדָ֥ר לְבוּשָׁ֑הּ eos וַ֝ תִּשְׂחַ֗ק לְ יֹ֣ום אַחֲרֹֽון׃ eov eos פִּ֭יהָ פָּתְחָ֣ה בְ חָכְמָ֑ה eos וְ תֹֽורַת־ חֶ֝֗סֶד עַל־ לְשֹׁונָֽהּ׃ eov eos צֹ֭ופִיָּה הֲלִיכֹ֣ות בֵּיתָ֑הּ eos וְ לֶ֥חֶם עַ֝צְל֗וּת לֹ֣א תֹאכֵֽל׃ eov eos קָ֣מוּ בָ֭נֶיהָ בַּ֝עְלָ֗הּ eos וַֽ יְאַשְּׁר֑וּהָ eos וַֽ יְהַֽלְלָֽהּ׃ eov eos רַבֹּ֣ות בָּ֭נֹות עָ֣שׂוּ חָ֑יִל eos וְ֝ אַ֗תְּ עָלִ֥ית עַל־ כֻּלָּֽנָה׃ eov eos שֶׁ֣קֶר הַ֭ חֵן eos וְ הֶ֣בֶל הַ יֹּ֑פִי eos אִשָּׁ֥ה יִרְאַת־ יְ֝הוָ֗ה הִ֣יא תִתְהַלָּֽל׃ eov eos תְּנוּ־ לָ֭הּ מִ פְּרִ֣י יָדֶ֑יהָ eos וִֽ יהַלְל֖וּהָ בַ שְּׁעָרִ֣ים מַעֲשֶֽׂיהָ׃
