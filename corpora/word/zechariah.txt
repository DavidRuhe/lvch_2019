eos בַּ יֹּ֣ום הַ ה֑וּא לֹֽא־ יִהְיֶ֣ה אֹ֔ור יְקָרֹ֖ות וְ קִפָּאֹֽון׃ eov eos וְ הָיָ֣ה eos יֹום־ אֶחָ֗ד ה֛וּא יִוָּדַ֥ע לַֽ יהוָ֖ה eos לֹא־ יֹ֣ום eos וְ לֹא־ לָ֑יְלָה eos וְ הָיָ֥ה eos לְ עֵֽת־ עֶ֖רֶב יִֽהְיֶה־ אֹֽור׃ eov eos וְ הָיָ֣ה׀ eos בַּ יֹּ֣ום הַ ה֗וּא יֵצְא֤וּ מַֽיִם־ חַיִּים֙ מִ יר֣וּשָׁלִַ֔ם חֶצְיָ֗ם אֶל־ הַ יָּם֙ הַ קַּדְמֹונִ֔י וְ חֶצְיָ֖ם אֶל־ הַ יָּ֣ם הָ אַחֲרֹ֑ון eos בַּ קַּ֥יִץ וּ בָ חֹ֖רֶף יִֽהְיֶֽה׃ eov eos וְ הָיָ֧ה יְהוָ֛ה לְ מֶ֖לֶךְ עַל־ כָּל־ הָ אָ֑רֶץ eos בַּ יֹּ֣ום הַ ה֗וּא יִהְיֶ֧ה יְהוָ֛ה אֶחָ֖ד וּ שְׁמֹ֥ו אֶחָֽד׃ eov eos יִסֹּ֨וב כָּל־ הָ אָ֤רֶץ כָּ עֲרָבָה֙ מִ גֶּ֣בַע לְ רִמֹּ֔ון נֶ֖גֶב יְרֽוּשָׁלִָ֑ם eos וְֽ רָאֲמָה֩ eos וְ יָשְׁבָ֨ה תַחְתֶּ֜יהָ לְ מִ שַּׁ֣עַר בִּנְיָמִ֗ן עַד־ מְקֹ֞ום שַׁ֤עַר הָֽ רִאשֹׁון֙ עַד־ שַׁ֣עַר הַ פִּנִּ֔ים וּ מִגְדַּ֣ל חֲנַנְאֵ֔ל עַ֖ד יִקְבֵ֥י הַ מֶּֽלֶךְ׃ eov eos וְ יָ֣שְׁבוּ בָ֔הּ eos וְ חֵ֖רֶם לֹ֣א יִֽהְיֶה־ עֹ֑וד eos וְ יָשְׁבָ֥ה יְרוּשָׁלִַ֖ם לָ בֶֽטַח׃ eov eos וְ זֹ֣את׀ תִּֽהְיֶ֣ה הַ מַּגֵּפָ֗ה אֲשֶׁ֨ר יִגֹּ֤ף יְהוָה֙ אֶת־ כָּל־ הָ֣ עַמִּ֔ים אֲשֶׁ֥ר צָבְא֖וּ עַל־ יְרוּשָׁלִָ֑ם eos הָמֵ֣ק׀ בְּשָׂרֹ֗ו eos וְ הוּא֙ עֹמֵ֣ד עַל־ רַגְלָ֔יו eos וְ עֵינָיו֙ תִּמַּ֣קְנָה בְ חֹֽרֵיהֶ֔ן eos וּ לְשֹׁונֹ֖ו תִּמַּ֥ק בְּ פִיהֶֽם׃ eov eos וְ הָיָה֙ eos בַּ יֹּ֣ום הַ ה֔וּא תִּֽהְיֶ֧ה מְהֽוּמַת־ יְהוָ֛ה רַבָּ֖ה בָּהֶ֑ם eos וְ הֶחֱזִ֗יקוּ אִ֚ישׁ יַ֣ד רֵעֵ֔הוּ eos וְ עָלְתָ֥ה יָדֹ֖ו עַל־ יַ֥ד רֵעֵֽהוּ׃ eov eos וְ גַ֨ם־ יְהוּדָ֔ה תִּלָּחֵ֖ם בִּ ירֽוּשָׁלִָ֑ם eos וְ אֻסַּף֩ חֵ֨יל כָּל־ הַ גֹּויִ֜ם סָבִ֗יב זָהָ֥ב וָ כֶ֛סֶף וּ בְגָדִ֖ים לָ רֹ֥ב מְאֹֽד׃ eov eos וְ כֵ֨ן תִּֽהְיֶ֜ה מַגֵּפַ֣ת הַ סּ֗וּס הַ פֶּ֨רֶד֙ הַ גָּמָ֣ל וְ הַ חֲמֹ֔ור וְ כָ֨ל־ הַ בְּהֵמָ֔ה אֲשֶׁ֥ר יִהְיֶ֖ה בַּ מַּחֲנֹ֣ות הָ הֵ֑מָּה eos כַּ מַּגֵּפָ֖ה הַ זֹּֽאת׃ eov eos וְ הָיָ֗ה eos כָּל־ הַ נֹּותָר֙ מִ כָּל־ הַ גֹּויִ֔ם הַ בָּאִ֖ים עַל־ יְרֽוּשָׁלִָ֑ם וְ עָל֞וּ מִ דֵּ֧י שָׁנָ֣ה בְ שָׁנָ֗ה לְ הִֽשְׁתַּחֲוֹת֙ לְ מֶ֨לֶךְ֙ יְהוָ֣ה צְבָאֹ֔ות וְ לָ חֹ֖ג אֶת־ חַ֥ג הַ סֻּכֹּֽות׃ eov eos וְ֠ הָיָה אֲשֶׁ֨ר לֹֽא־ יַעֲלֶ֜ה מֵ אֵ֨ת מִשְׁפְּחֹ֤ות הָ אָ֨רֶץ֙ אֶל־ יְר֣וּשָׁלִַ֔ם לְ הִֽשְׁתַּחֲוֹ֔ת לְ מֶ֖לֶךְ יְהוָ֣ה צְבָאֹ֑ות וְ לֹ֥א עֲלֵיהֶ֖ם יִהְיֶ֥ה הַ גָּֽשֶׁם׃ eov eos וְ אִם־ מִשְׁפַּ֨חַת מִצְרַ֧יִם לֹֽא־ תַעֲלֶ֛ה eos וְ לֹ֥א בָאָ֖ה eos וְ לֹ֣א עֲלֵיהֶ֑ם תִּֽהְיֶ֣ה הַ מַּגֵּפָ֗ה אֲשֶׁ֨ר יִגֹּ֤ף יְהוָה֙ אֶת־ הַ גֹּויִ֔ם אֲשֶׁר֙ לֹ֣א יַֽעֲל֔וּ לָ חֹ֖ג אֶת־ חַ֥ג הַ סֻּכֹּֽות׃ eov eos זֹ֥את תִּהְיֶ֖ה חַטַּ֣את מִצְרָ֑יִם וְ חַטַּאת֙ כָּל־ הַ גֹּויִ֔ם אֲשֶׁר֙ לֹ֣א יַֽעֲל֔וּ לָ חֹ֖ג אֶת־ חַ֥ג הַ סֻּכֹּֽות׃ eov eos בַּ יֹּ֣ום הַ ה֗וּא יִֽהְיֶה֙ עַל־ מְצִלֹּ֣ות הַ סּ֔וּס קֹ֖דֶשׁ לַֽ יהוָ֑ה eos וְ הָיָ֤ה הַ סִּירֹות֙ בְּ בֵ֣ית יְהוָ֔ה כַּ מִּזְרָקִ֖ים לִ פְנֵ֥י הַ מִּזְבֵּֽחַ׃ eov eos וְ֠ הָיָה כָּל־ סִ֨יר בִּ ירוּשָׁלִַ֜ם וּ בִֽ יהוּדָ֗ה קֹ֚דֶשׁ לַ יהוָ֣ה צְבָאֹ֔ות eos וּ בָ֨אוּ֙ כָּל־ הַ זֹּ֣בְחִ֔ים eos וְ לָקְח֥וּ מֵהֶ֖ם eos וּ
