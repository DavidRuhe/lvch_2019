אֶהֱרֹ֑ג eos לֹֽא־ יָנ֤וּס לָהֶם֙ נָ֔ס eos וְ לֹֽא־ יִמָּלֵ֥ט לָהֶ֖ם פָּלִֽיט׃ eov eos אִם־ יַחְתְּר֣וּ בִ שְׁאֹ֔ול eos מִ שָּׁ֖ם יָדִ֣י תִקָּחֵ֑ם eos וְ אִֽם־ יַעֲלוּ֙ הַ שָּׁמַ֔יִם eos מִ שָּׁ֖ם אֹורִידֵֽם׃ eov eos וְ אִם־ יֵחָֽבְאוּ֙ בְּ רֹ֣אשׁ הַ כַּרְמֶ֔ל eos מִ שָּׁ֥ם אֲחַפֵּ֖שׂ eos וּ לְקַחְתִּ֑ים eos וְ אִם־ יִסָּ֨תְר֜וּ מִ נֶּ֤גֶד עֵינַי֙ בְּ קַרְקַ֣ע הַ יָּ֔ם eos מִ שָּׁ֛ם אֲצַוֶּ֥ה אֶת־ הַ נָּחָ֖שׁ eos וּ נְשָׁכָֽם׃ eov eos וְ אִם־ יֵלְכ֤וּ בַ שְּׁבִי֙ לִ פְנֵ֣י אֹֽיבֵיהֶ֔ם eos מִ שָּׁ֛ם אֲצַוֶּ֥ה אֶת־ הַ חֶ֖רֶב eos וַ הֲרָגָ֑תַם eos וְ שַׂמְתִּ֨י עֵינִ֧י עֲלֵיהֶ֛ם לְ רָעָ֖ה וְ לֹ֥א לְ טֹובָֽה׃ eov eos הֲ לֹ֣וא כִ בְנֵי֩ כֻשִׁיִּ֨ים אַתֶּ֥ם לִ֛י בְּנֵ֥י יִשְׂרָאֵ֖ל eos נְאֻם־ יְהוָ֑ה eos הֲ לֹ֣וא אֶת־ יִשְׂרָאֵ֗ל הֶעֱלֵ֨יתִי֙ מֵ אֶ֣רֶץ מִצְרַ֔יִם וּ פְלִשְׁתִּיִּ֥ים מִ כַּפְתֹּ֖ור וַ אֲרָ֥ם מִ קִּֽיר׃ eov eos הִנֵּ֞ה עֵינֵ֣י׀ אֲדֹנָ֣י יְהוִ֗ה בַּ מַּמְלָכָה֙ הַֽ חַטָּאָ֔ה eos וְ הִשְׁמַדְתִּ֣י אֹתָ֔הּ מֵ עַ֖ל פְּנֵ֣י הָ אֲדָמָ֑ה eos אֶ֗פֶס כִּ֠י לֹ֣א הַשְׁמֵ֥יד אַשְׁמִ֛יד אֶת־ בֵּ֥ית יַעֲקֹ֖ב eos נְאֻם־ יְהוָֽה׃ eov eos כִּֽי־ הִנֵּ֤ה אָֽנֹכִי֙ מְצַוֶּ֔ה eos וַ הֲנִעֹ֥ותִי בְ כָֽל־ הַ גֹּויִ֖ם אֶת־ בֵּ֣ית יִשְׂרָאֵ֑ל כַּ אֲשֶׁ֤ר יִנֹּ֨ועַ֙ בַּ כְּבָרָ֔ה eos וְ לֹֽא־ יִפֹּ֥ול צְרֹ֖ור אָֽרֶץ׃ eov eos בַּ חֶ֣רֶב יָמ֔וּתוּ כֹּ֖ל חַטָּאֵ֣י עַמִּ֑י הָ אֹמְרִ֗ים eos לֹֽא־ תַגִּ֧ישׁ eos וְ תַקְדִּ֛ים בַּעֲדֵ֖ינוּ הָ רָעָֽה׃ eov eos בַּ יֹּ֣ום הַ ה֔וּא אָקִ֛ים אֶת־ סֻכַּ֥ת דָּוִ֖יד הַ נֹּפֶ֑לֶת eos וְ גָדַרְתִּ֣י אֶת־ פִּרְצֵיהֶ֗ן eos וַ הֲרִֽסֹתָיו֙ אָקִ֔ים eos וּ בְנִיתִ֖יהָ כִּ ימֵ֥י עֹולָֽם׃ eov eos לְמַ֨עַן יִֽירְשׁ֜וּ אֶת־ שְׁאֵרִ֤ית אֱדֹום֙ וְ כָל־ הַ גֹּויִ֔ם אֲשֶׁר־ נִקְרָ֥א שְׁמִ֖י עֲלֵיהֶ֑ם eos נְאֻם־ יְהוָ֖ה עֹ֥שֶׂה זֹּֽאת׃ פ eov eos הִנֵּ֨ה יָמִ֤ים בָּאִים֙ eos נְאֻם־ יְהוָ֔ה eos וְ נִגַּ֤שׁ חֹורֵשׁ֙ בַּ קֹּצֵ֔ר וְ דֹרֵ֥ךְ עֲנָבִ֖ים בְּ מֹשֵׁ֣ךְ הַ זָּ֑רַע eos וְ הִטִּ֤יפוּ הֶֽ הָרִים֙ עָסִ֔יס eos וְ כָל־ הַ גְּבָעֹ֖ות תִּתְמֹוגַֽגְנָה׃ eov eos וְ שַׁבְתִּי֮ אֶת־ שְׁב֣וּת עַמִּ֣י יִשְׂרָאֵל֒ eos וּ בָנ֞וּ עָרִ֤ים נְשַׁמֹּות֙ eos וְ יָשָׁ֔בוּ eos וְ נָטְע֣וּ כְרָמִ֔ים eos וְ שָׁת֖וּ אֶת־ יֵינָ֑ם eos וְ עָשׂ֣וּ גַנֹּ֔ות eos וְ אָכְל֖וּ אֶת־ פְּרִיהֶֽם׃ eov eos וּ נְטַעְתִּ֖ים עַל־ אַדְמָתָ֑ם eos וְ לֹ֨א יִנָּתְשׁ֜וּ עֹ֗וד מֵ עַ֤ל אַדְמָתָם֙ אֲשֶׁ֣ר נָתַ֣תִּי לָהֶ֔ם eos אָמַ֖ר יְהוָ֥ה אֱלֹהֶֽיךָ׃
